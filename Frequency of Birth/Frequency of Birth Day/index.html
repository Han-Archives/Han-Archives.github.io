def yearly_births(df,year_num):
    df = df[df['year'] == year_num]
    df = df.sort_values('births', ascending=False) # 최대값, 최소값 확인
    
    df_pivot = df.pivot('date','month','births')
    df_pivot = df_pivot.fillna(0)
    df_pivot = df_pivot.astype('int')
    
    return df, df_pivot
